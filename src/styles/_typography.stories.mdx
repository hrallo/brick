import { Meta, Source, Story, Canvas, ArgsTable } from '@storybook/addon-docs'; 
import { loremIpsum } from 'lorem-ipsum'
import * as R from 'ramda'

<Meta title="Styles/Typography" />

export const opts = {
    'body': ['body1', 'body2'],
    'headings': ['heading1', 'heading2', 'heading3', 'subheading1','subheading2', 'subheading3'],
    'link': ['link', 'link-large', 'link-small'],
    'misc': ['button', 'caps', 'caps-small', 'code', 'pill'],
    'utils': ['text-transform-uppercase', 'text-transform-lowercase', 'text-transform-capitalize', 'text-align-left', 'text-align-center', 'text-align-right'],
}

export const copyShort = loremIpsum({ count: 3, units: 'words' })
export const copyLong = loremIpsum({ count: 1, units: 'paragraphs' })
export const Template = (args) => <div {...args}>{args.copy}</div>;
